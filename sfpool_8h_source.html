47// Memory pool structure
48typedef struct sfpool_t {
49 uint8_t *buffer; // raw
50 uint8_t *data; // aligned
51 uint8_t *free_list;
52 uint32_t free_count;
53 uint32_t total_blocks;
54 uint32_t total_bytes;
55 uint32_t block_size;
56#ifdef PROFILING
57 uint32_t *hits;
58 uint32_t hits_total;
59 size_t hits_bytes;
60 uint32_t miss_total;
61 size_t miss_bytes;
62 size_t alloc_total;
63#endif
64} sfpool_t;

66#if defined(__x86_64__) || defined(_M_X64) || defined(__ppc64__) || defined(__LP64__)
67#define ptr_t uint64_t
68#define ptr_align 8
69#else
70#define ptr_t uint32_t
71#define ptr_align 4
72#endif
73#if !defined(__MUSL__)
74static_assert(sizeof(ptr_t) == sizeof(void*), "Unknown memory pointer size detected");

76static inline bool _is_in_pool(sfpool_t *pool, const void *ptr) {
77 volatile ptr_t p = (ptr_t)ptr;
78 return(p >= (ptr_t)pool->data
79 && p < (ptr_t)(pool->data + pool->total_bytes));
80}

95void sfutil_zero(void *ptr, uint32_t size) {
96 register uint32_t *p = (uint32_t*)ptr; // use 32bit pointer
97 register uint32_t s = (size>>2); // divide counter by 4
98 while (s--) *p++ = 0x0; // hit the road jack
99}

109void *sfutil_memalign(const void* ptr) {
110 register ptr_t mask = ptr_align - 1;
111 ptr_t aligned = ((ptr_t)ptr + mask) & ~mask;
112 return (void*)aligned;
113}

123void *sfutil_secalloc(size_t size) {
124 // add bytes to every allocation to support alignment
125 void *res = NULL;
126#if defined(__EMSCRIPTEN__)
127 res = (uint8_t *)malloc(size+ptr_align);
128#elif defined(_WIN32)
129 res = VirtualAlloc(NULL, size+ptr_align,
130 MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
131#elif defined(__APPLE__)
132 int flags = MAP_PRIVATE | MAP_ANONYMOUS | MAP_NORESERVE;
133 res = mmap(NULL, size, PROT_READ | PROT_WRITE, flags, -1, 0);
134 mlock(res, size);
135#else // assume POSIX
136 int flags = MAP_PRIVATE | MAP_ANONYMOUS | MAP_NORESERVE;
137 struct rlimit rl;
138 if (getrlimit(RLIMIT_MEMLOCK, &rl) == 0)
139 if(size<=rl.rlim_cur) flags |= MAP_LOCKED;
140 res = mmap(NULL, size, PROT_READ | PROT_WRITE, flags, -1, 0);
141#endif
142 return res;
143}

153void sfutil_secfree(void *ptr, size_t size) {
154#if defined(__EMSCRIPTEN__)
155 free(ptr);
156#elif defined(_WIN32)
157 VirtualFree(ptr, 0, MEM_RELEASE);
158#else // Posix
159 munmap(ptr, size +ptr_align);
160#endif
161}

182size_t sfpool_init(sfpool_t *pool, size_t nmemb, size_t blocksize) {
183 if((blocksize & (blocksize - 1)) != 0) return 0;
184 // SFPool blocksize must be a power of two
185 size_t totalsize = nmemb * blocksize;
186 pool->buffer = sfutil_secalloc(totalsize);
187 if (pool->buffer == NULL) return 0;
188 // Failed to allocate pool memory
189 pool->data = sfutil_memalign(pool->buffer);
190 if (pool->data == NULL) return 0;
191 // Failed to allocate pool memory
192 pool->total_bytes = totalsize;
193 pool->total_blocks = nmemb;
194 pool->block_size = blocksize;
195 // Initialize the embedded free list
196 pool->free_list = pool->data;
197 register uint32_t i, bi;
198 for (i = 0; i < pool->total_blocks - 1; ++i) {
199 bi = i*blocksize;
200 *(uint8_t **)(pool->data + bi) =
201 pool->data + bi + blocksize;
202 }
203 pool->free_count = pool->total_blocks;
204 *(uint8_t **)
205 (pool->data + (pool->total_blocks - 1) * blocksize) = NULL;
206#ifdef PROFILING
207 pool->miss_total = pool->miss_bytes = 0;
208 pool->hits_total = pool->hits_bytes = 0;
209 pool->alloc_total = 0;
210#endif
211 return totalsize;
212}

242void *sfpool_malloc(void *restrict opaque, const size_t size) {
243 sfpool_t *pool = (sfpool_t*)opaque;
244 void *ptr;
245 if (size <= pool->block_size
246 && pool->free_list != NULL) {
247#ifdef PROFILING
248 pool->hits_total++;
249 pool->hits_bytes+=size;
250 pool->alloc_total+=size;
251#endif
252 // Remove the first block from the free list
253 uint8_t *block = pool->free_list;
254 pool->free_list = *(uint8_t **)block;
255 pool->free_count-- ;
256 return block;
257 }
258 // Fallback to system malloc for large allocations
259 ptr = malloc(size);
260 if(ptr == NULL) perror("system malloc error");
261#ifdef PROFILING
262 pool->miss_total++;
263 pool->miss_bytes+=size;
264 pool->alloc_total+=size;
265#endif
266 return ptr;
267}

279void sfpool_free(void *restrict opaque, void *ptr) {
280 sfpool_t *pool = (sfpool_t*)opaque;
281 if (ptr == NULL) return; // Freeing NULL is a no-op
282 if (_is_in_pool(pool,ptr)) {
283 // Add the block back to the free list
284 *(uint8_t **)ptr = pool->free_list;
285 pool->free_list = (uint8_t *)ptr;
286 pool->free_count++ ;
287#ifdef SECURE_ZERO
288 // Zero out the block for security
289 sfutil_zero(ptr, pool->block_size);
290#endif
291 return;
292 } else {
293#ifdef FALLBACK
294 free(ptr);
295#endif
296 }
297}

311void *sfpool_realloc(void *restrict opaque, void *ptr, const size_t size) {
312 sfpool_t *pool = (sfpool_t*)opaque;
313 if (ptr == NULL) {
314 return sfpool_malloc(pool, size);
315 }
316 if (size == 0) {
317 sfpool_free(pool, ptr);
318 return NULL;
319 }
320 if (_is_in_pool((sfpool_t*)pool,ptr)) {
321 if (size <= pool->block_size) {
322#ifdef PROFILING
323 pool->hits_total++;
324 pool->hits_bytes+=size;
325 pool->alloc_total+=size;
326#endif
327 return ptr; // No need to reallocate
328 } else {
329 void *new_ptr = malloc(size);
330 memcpy(new_ptr, ptr, pool->block_size); // Copy only BLOCK_SIZE bytes
331#ifdef SECURE_ZERO
332 sfutil_zero(ptr, pool->block_size); // Zero out the old block
333#endif
334 // Add the block back to the free list
335 *(uint8_t **)ptr = pool->free_list;
336 pool->free_list = (uint8_t *)ptr;
337 pool->free_count++ ;
338#ifdef SECURE_ZERO
339 // Zero out the block for security
340 sfutil_zero(ptr, pool->block_size);
341#endif
342#ifdef PROFILING
343 pool->miss_total++;
344 pool->miss_bytes+=size;
345 pool->alloc_total+=size;
346#endif
347 return new_ptr;
348 }
349 } else {
350#ifdef FALLBACK
351 // Handle large allocations
352 return realloc(ptr, size);
353#ifdef PROFILING
354 pool->miss_total++;
355 pool->miss_bytes+=size;
356 pool->alloc_total+=size;
357#endif
358#else
359 return NULL;
360#endif
361 }
362}
